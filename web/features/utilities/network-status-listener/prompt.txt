import { useEffect, useRef } from "react";
import { useNetworkState } from "@uidotdev/usehooks";
import { toast } from "sonner";

interface NetworkStatusListenerProps {
  offlineMessage?: string;
  onlineMessage?: string;
  showToast?: boolean;
}

/**
 * Monitors network connectivity and shows toast notifications on status changes.
 */
export function NetworkStatusListener({
  offlineMessage = "You are offline. Please check your connection.",
  onlineMessage = "Connection restored. You are back online.",
  showToast = true,
}: NetworkStatusListenerProps) {
  const network = useNetworkState();
  const wasOffline = useRef(false);
  const isFirstRender = useRef(true);

  useEffect(() => {
    if (isFirstRender.current) {
      isFirstRender.current = false;
      wasOffline.current = !network.online;
      return;
    }

    if (!network.online && !wasOffline.current) {
      wasOffline.current = true;
      if (showToast) {
        toast.error(offlineMessage, { duration: 5000 });
      }
    } else if (network.online && wasOffline.current) {
      wasOffline.current = false;
      if (showToast) {
        toast.success(onlineMessage, { duration: 3000 });
      }
    }
  }, [network.online, showToast, offlineMessage, onlineMessage]);

  return null;
}

Example integration:

```tsx
import dynamic from "next/dynamic";
import { Toaster } from "sonner";

const NetworkStatusListener = dynamic(
  () => import("@/components/network-status-listener").then((mod) => ({
    default: mod.NetworkStatusListener,
  })),
  { ssr: false }
);

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <html lang="en">
      <body>
        {children}
        <NetworkStatusListener />
        <Toaster richColors position="top-right" />
      </body>
    </html>
  );
}
```

Important notes:
- The component must be imported with dynamic import and ssr: false because it uses browser APIs
- The Toaster component from sonner is required for toast notifications to work
- The component returns null, so it doesn't render any UI - it only shows toast notifications
- Customize messages by passing offlineMessage and onlineMessage props
- Disable toasts by setting showToast={false}
